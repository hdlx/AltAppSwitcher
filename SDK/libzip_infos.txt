// Zlib an libzip built from source using mingw UCRT

// CMake params:

// zlib:
cmake -G "MinGW Makefiles" -D CMAKE_C_COMPILER="aarch64-w64-mingw32-clang.exe" -D CMAKE_MAKE_PROGRAM="mingw32-make.exe"

cmake --build . --target zlibstatic 

// libzip
cmake -G "MinGW Makefiles" -D CMAKE_C_COMPILER="aarch64-w64-mingw32-clang.exe" -D CMAKE_MAKE_PROGRAM="mingw32-make.exe"  -D BUILD_SHARED_LIBS="OFF" -D ZLIB_INCLUDE_DIR="path/to/zlib-1.3.1" -D ZLIB_LIBRARY="path/to/zlib-1.3.1/libzlibstatic.a" 

cmake --build .
